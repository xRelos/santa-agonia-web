<script>
(function(){

  /* ============================================================
     ▼▼▼  AQUÍ SE CAMBIAN LOS ENLACES  ▼▼▼
     col: 1 → columna "La Agrupación"
     col: 2 → columna "Más información"
  ============================================================ */
  var L=[
    {label:"Inicio",     href:"index.html",     col:1},
    {label:"Historia",   href:"historia.html",   col:1},
    {label:"Agonía",     href:"agonia.html",     col:1},
    {label:"Condena",    href:"condena.html",    col:1},
    {label:"Vera Cruz",  href:"vera.html",       col:1},
    {label:"Actos",      href:"actos.html",      col:2},
    {label:"Multimedia", href:"multimedia.html", col:2},
    {label:"Directiva",  href:"directiva.html",  col:2},
    {label:"Contacto",   href:"contacto.html",   col:2},
  ];
  /* ▲▲▲  FIN  ▲▲▲ */

  var n1=document.getElementById('sf-nav1');
  var n2=document.getElementById('sf-nav2');
  L.forEach(function(l){
    var li=document.createElement('li');
    var a=document.createElement('a');
    a.href=l.href;a.textContent=l.label;
    li.appendChild(a);
    (l.col===1?n1:n2).appendChild(li);
  });

  document.getElementById('sf-year').textContent=new Date().getFullYear();

  /* Acordeón móvil */
  document.querySelectorAll('#sf .sf-col h4').forEach(function(h4){
    h4.addEventListener('click',function(){
      var col=h4.closest('.sf-col');
      var wasOpen=col.classList.contains('open');
      document.querySelectorAll('#sf .sf-col').forEach(function(c){c.classList.remove('open');});
      if(!wasOpen)col.classList.add('open');
    });
  });

})();
</script>
