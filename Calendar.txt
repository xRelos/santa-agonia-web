        // CARGA DE EVENTOS DESDE GOOGLE CALENDAR
        // ============================================================
        async function loadGoogleCalendarEvents() {
            try {
                const timeMin = new Date();
                timeMin.setMonth(timeMin.getMonth() - 2);
                const timeMax = new Date();
                timeMax.setMonth(timeMax.getMonth() + 6);

                const url = `https://www.googleapis.com/calendar/v3/calendars/${encodeURIComponent(CALENDAR_ID)}/events?key=${GOOGLE_API_KEY}&timeMin=${timeMin.toISOString()}&timeMax=${timeMax.toISOString()}&singleEvents=true&orderBy=startTime`;

                const response = await fetch(url);
                if (!response.ok) {
                    throw new Error('Error al cargar eventos');
                }

                const data = await response.json();
                const events = [];
                const originalEvents = [];

                data.items.forEach(item => {
                    // Detectamos la categoría según el texto del evento
                    let category = 'Agonía';
                    const description = item.description || '';
                    const summary = item.summary || '';
                    const searchText = (description + ' ' + summary).toLowerCase();

                    if (searchText.includes('condena')) {
                        category = 'Condena';
                    } else if (searchText.includes('vera') || searchText.includes('cruz')) {
                        category = 'Vera Cruz';
                    } else if (searchText.includes('marrajos') || searchText.includes('cofradia') || searchText.includes('cofradía')) {
                        category = 'Marrajos/Cofradía';
                    } else if (searchText.includes('agonia') || searchText.includes('agonía')) {
                        category = 'Agonía';
                    }

                    const startDate = item.start.dateTime || item.start.date;
                    const endDate = item.end.dateTime || item.end.date;
                    const isAllDay = !item.start.dateTime;

                    originalEvents.push({
                        id: item.id,
                        title: item.summary || 'Sin título',
                        start: startDate,
                        end: endDate,
                        allDay: isAllDay,
                        description: description,
                        location: item.location || '',
                        category: category,
                        htmlLink: item.htmlLink
                    });

                    // Si es un evento de varios días, lo dividimos en días individuales
                    const start = new Date(startDate);
                    const end = new Date(endDate);
                    const diffDays = Math.ceil((end - start) / (1000 * 60 * 60 * 24));

                    if (diffDays > 1) {
                        for (let i = 0; i < diffDays; i++) {
                            const currentDay = new Date(start);
                            currentDay.setDate(start.getDate() + i);
                            const nextDay = new Date(currentDay);
                            nextDay.setDate(currentDay.getDate() + 1);

                            events.push({
                                id: item.id + '_day' + i,
                                title: item.summary || 'Sin título',
                                start: (i === 0 && !isAllDay) ? startDate : currentDay.toISOString().split('T')[0],
                                end: (i === diffDays - 1 && !isAllDay) ? endDate : nextDay.toISOString().split('T')[0],
                                allDay: isAllDay || i > 0,
                                description: description,
                                location: item.location || '',
                                category: category,
                                htmlLink: item.htmlLink,
                                originalId: item.id,
                                isDuplicate: i > 0
                            });
                        }
                    } else {
                        events.push({
                            id: item.id,
                            title: item.summary || 'Sin título',
                            start: startDate,
                            end: endDate,
                            allDay: isAllDay,
                            description: description,
                            location: item.location || '',
                            category: category,
                            htmlLink: item.htmlLink
                        });
                    }
                });

                calendar.removeAllEvents();
                calendar.addEventSource(events);

                allEvents = originalEvents;

                if (currentView === 'list') {
                    renderListView();
                }

                setTimeout(forceSquareCells, 200);

                if (isMobile) {
                    setTimeout(applyMobileDotStyles, 300);
                }

            } catch (error) {
                console.error('Error loading events:', error);
                alert('Error al cargar los eventos del calendario. Por favor, intenta de nuevo más tarde.');
            }
        }

        function shareEvent() {
            if (!currentEvent) return;

            const startDate = new Date(currentEvent.start);
            const dateStr = startDate.toLocaleDateString('es-ES', {
                year: 'numeric',
                month: 'long',
                day: 'numeric'
            });
            const timeStr = startDate.toLocaleTimeString('es-ES', {
                hour: '2-digit',
                minute: '2-digit'
            });

            const shareText = `${currentEvent.title}\n${dateStr} a las ${timeStr}`;

            if (navigator.share) {
                navigator.share({
                    title: currentEvent.title,
                    text: shareText,
                    url: window.location.href
                }).catch(err => console.log('Error sharing:', err));
            } else {
                navigator.clipboard.writeText(shareText).then(() => {
                    alert('Información del evento copiada al portapapeles');
                }).catch(err => {
                    console.error('Error copying to clipboard:', err);
                });
            }
        }

                calendarBtn.classList.remove('active');
                listBtn.classList.add('active');
                renderListView();
            }
        }

        // VISTA DE LISTA
        // ============================================================

        function escapeHtml(text) {
            if (!text) return '';
            const div = document.createElement('div');
            div.textContent = text;
            return div.innerHTML;
        }

        function renderListView() {
            const listContainer = document.getElementById('listEvents');

            if (allEvents.length === 0) {
                listContainer.innerHTML = '<div class="loading"><p>No hay eventos disponibles</p></div>';
                return;
            }

            // Filtramos eventos del mes actual
            const filteredEvents = allEvents.filter(event => {
                const eventStart = new Date(event.start);
                const eventEnd = event.end ? new Date(event.end) : eventStart;
                const monthStart = new Date(currentListYear, currentListMonth, 1);
                const monthEnd = new Date(currentListYear, currentListMonth + 1, 0);
                return (eventStart <= monthEnd && eventEnd >= monthStart);
            });

            if (filteredEvents.length === 0) {
                const monthDate = new Date(currentListYear, currentListMonth, 1);
                const monthName = monthDate.toLocaleDateString('es-ES', { month: 'long', year: 'numeric' });
                const capitalizedMonth = monthName.charAt(0).toUpperCase() + monthName.slice(1);

                listContainer.innerHTML = `
                    <div class="list-month-header">
                        <div class="list-month-title">${capitalizedMonth}</div>
                        <div class="list-month-nav">
                            <button class="list-month-nav-button" onclick="changeListMonth(-1)">‹</button>
                            <button class="list-month-nav-button" onclick="changeListMonth(0)">Hoy</button>
                            <button class="list-month-nav-button" onclick="changeListMonth(1)">›</button>
                        </div>
                    </div>
                    <div class="loading"><p>No hay eventos en este mes</p></div>
                `;
                return;
            }

            const sortedEvents = [...filteredEvents].sort((a, b) => {
                return new Date(a.start) - new Date(b.start);
            });

            const monthDate = new Date(currentListYear, currentListMonth, 1);
            const monthName = monthDate.toLocaleDateString('es-ES', { month: 'long', year: 'numeric' });
            const capitalizedMonth = monthName.charAt(0).toUpperCase() + monthName.slice(1);

            let html = `
                <div class="list-month-header">
                    <div class="list-month-title">${capitalizedMonth}</div>
                    <div class="list-month-nav">
                        <button class="list-month-nav-button" onclick="changeListMonth(-1)">‹</button>
                        <button class="list-month-nav-button" onclick="changeListMonth(0)">Hoy</button>
                        <button class="list-month-nav-button" onclick="changeListMonth(1)">›</button>
                    </div>
                </div>
                <div class="list-month-events">
            `;

            html += sortedEvents.map(event => {
                const startDate = new Date(event.start);
                const endDate = event.end ? new Date(event.end) : startDate;
                const isMultiDay = event.allDay && (endDate - startDate) > (24 * 60 * 60 * 1000);

                let dateStr;
                if (isMultiDay) {
                    const startDay = startDate.getDate();
                    const endDay = endDate.getDate() - 1;
                    const startMonth = startDate.toLocaleDateString('es-ES', { month: 'long' });
                    const endMonth = new Date(endDate.getTime() - 86400000).toLocaleDateString('es-ES', { month: 'long' });
                    const year = startDate.getFullYear();

                    if (startMonth === endMonth) {
                        dateStr = `Del ${startDay} al ${endDay} de ${startMonth} de ${year}`;
                    } else {
                        dateStr = `Del ${startDay} de ${startMonth} al ${endDay} de ${endMonth} de ${year}`;
                    }
                } else {
                    const dateOptions = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
                    dateStr = startDate.toLocaleDateString('es-ES', dateOptions);
                }

                let timeStr = '';
                if (!event.allDay) {
                    const timeOptions = { hour: '2-digit', minute: '2-digit' };
                    timeStr = startDate.toLocaleTimeString('es-ES', timeOptions) + ' - ' +
                              endDate.toLocaleTimeString('es-ES', timeOptions);
                }

                let category = event.category || 'Agonía';
                let categoryClass = 'event-agonia';
                if (categoryColors[category.toLowerCase()]) {
                    categoryClass = categoryColors[category.toLowerCase()];
                } else if (categoryColors[category]) {
                    categoryClass = categoryColors[category];
                }

                let displayCategory = category;
                if (category.toLowerCase().includes('agonia') || category === 'Agonía') {
                    displayCategory = 'Agonía';
                } else if (category.toLowerCase().includes('condena')) {
                    displayCategory = 'Condena';
                } else if (category.toLowerCase().includes('vera') || category.toLowerCase().includes('cruz')) {
                    displayCategory = 'Vera Cruz';
                } else if (category.toLowerCase().includes('marrajos') || category.toLowerCase().includes('cofradia')) {
                    displayCategory = 'Marrajos/Cofradía';
                }

                const eventTitle = escapeHtml(event.title);
                const eventLocation = escapeHtml(event.location || '');
                const eventId = escapeHtml(event.id);

                return `
                    <div class="list-event-item ${categoryClass}" data-event-id="${eventId}">
                        <div class="list-event-header">
                            <div class="list-event-title">${eventTitle}</div>
                            <span class="list-event-category ${categoryClass}">${displayCategory}</span>
                        </div>
                        <div class="list-event-details">
                            <div class="list-event-date">
                                <svg class="list-icon" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24" stroke="currentColor">
                                    <rect x="3" y="4" width="18" height="18" rx="2" ry="2"></rect>
                                    <line x1="16" y1="2" x2="16" y2="6"></line>
                                    <line x1="8" y1="2" x2="8" y2="6"></line>
                                    <line x1="3" y1="10" x2="21" y2="10"></line>
                                </svg>
                                <span>${dateStr}</span>
                            </div>
                            ${timeStr ? `
                            <div class="list-event-time">
                                <svg class="list-icon" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24" stroke="currentColor">
                                    <circle cx="12" cy="12" r="10"></circle>
                                    <polyline points="12 6 12 12 16 14"></polyline>
                                </svg>
                                <span>${timeStr}</span>
                            </div>
                            ` : ''}
                            ${eventLocation ? `
                            <div class="list-event-location">
                                <svg class="list-icon" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24" stroke="currentColor">
                                    <path d="M21 10c0 7-9 13-9 13s-9-6-9-13a9 9 0 0 1 18 0z"></path>
                                    <circle cx="12" cy="10" r="3"></circle>
                                </svg>
                                <span>${eventLocation}</span>
                            </div>
                            ` : ''}
                        </div>
                    </div>
                `;
            }).join('');

            html += '</div>';
            listContainer.innerHTML = html;

            // Añadir eventos de clic a los elementos de la lista
            listContainer.querySelectorAll('.list-event-item').forEach((item) => {
                const eventId = item.getAttribute('data-event-id');
                const event = sortedEvents.find(e => e.id === eventId);
                if (event) {
                    item.addEventListener('click', function() {
                        showEventFromList(event);
                    });
                }
            });
        }

        // ============================================================
        // NAVEGACIÓN DE MESES EN LA VISTA DE LISTA
        // ============================================================

        function changeListMonth(direction) {
            if (direction === 0) {
                const now = new Date();
                currentListMonth = now.getMonth();
                currentListYear = now.getFullYear();
            } else {
                currentListMonth += direction;
                if (currentListMonth < 0) {
                    currentListMonth = 11;
                    currentListYear--;
                } else if (currentListMonth > 11) {
                    currentListMonth = 0;
                    currentListYear++;
                }
            }
            renderListView();
        }

        function showEventFromList(event) {
            const mockEvent = {
                id: event.id,
                title: event.title,
                start: event.start,
                end: event.end,
                extendedProps: {
                    category: event.category,
                    location: event.location,
                    description: event.description,
                    htmlLink: event.htmlLink
                }
            };
            showEventModal(mockEvent);
        }
