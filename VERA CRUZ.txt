let currentItem = 'imagen';
let currentImageIndex = 0;

const modal = document.getElementById('patrimonioModal');
const modalTitle = document.getElementById('modalTitle');
const modalHeader = document.getElementById('modalHeader');
const modalContent = document.getElementById('modalContent');
const carouselImage = document.getElementById('carouselImage');
const imageCaption = document.getElementById('imageCaption');
const carouselIndicators = document.getElementById('carouselIndicators');
const closeModal = document.getElementById('closeModal');
const prevBtn = document.getElementById('prevBtn');
const nextBtn = document.getElementById('nextBtn');

function openModal(itemId) {
    currentItem = itemId;
    currentImageIndex = 0;
    const data = patrimonioData[itemId];
    modalTitle.textContent = data.title;
    modalContent.innerHTML = data.content;
    modalHeader.className = 'flex items-center justify-between p-6 border-b border-gray-200 dark:border-gray-800 bg-gradient-to-r from-veracruz to-veracruz-dark text-white';
    updateCarousel();
    modal.classList.add('active');
    document.body.style.overflow = 'hidden';
}

function closeModalFunc() {
    modal.classList.remove('active');
    document.body.style.overflow = 'auto';
}

function updateCarousel() {
    const data = patrimonioData[currentItem];
    const imageData = data.images[currentImageIndex];
    carouselImage.src = imageData.src;
    carouselImage.alt = imageData.caption;
    imageCaption.textContent = imageData.caption;
    carouselIndicators.innerHTML = '';
    data.images.forEach((_, index) => {
        const indicator = document.createElement('div');
        indicator.className = `w-3 h-3 rounded-full cursor-pointer ${index === currentImageIndex ? 'bg-veracruz' : 'bg-gray-300 dark:bg-gray-700'}`;
        indicator.addEventListener('click', () => { currentImageIndex = index; updateCarousel(); });
        carouselIndicators.appendChild(indicator);
    });
}

function nextImage() {
    const data = patrimonioData[currentItem];
    currentImageIndex = (currentImageIndex + 1) % data.images.length;
    updateCarousel();
}

function prevImage() {
    const data = patrimonioData[currentItem];
    currentImageIndex = (currentImageIndex - 1 + data.images.length) % data.images.length;
    updateCarousel();
}

document.querySelectorAll('.modern-bubble').forEach(bubble => {
    bubble.addEventListener('click', () => {
        const target = bubble.getAttribute('data-target');
        openModal(target);
        bubble.style.transform = 'scale(0.95)';
        setTimeout(() => { bubble.style.transform = ''; }, 150);
    });
});

closeModal.addEventListener('click', closeModalFunc);
prevBtn.addEventListener('click', prevImage);
nextBtn.addEventListener('click', nextImage);
modal.addEventListener('click', (e) => { if (e.target === modal) closeModalFunc(); });
document.addEventListener('keydown', (e) => { if (e.key === 'Escape' && modal.classList.contains('active')) closeModalFunc(); });
</script>
