  // ============================================================
  // FUNCIONES PARA ABRIR Y CERRAR MODALES

  // Abre el submodal con la categoría seleccionada (revistas o historicos)
  function openSubModal(category) {
    // Elegimos el array de datos según la categoría
    const items = category === 'revistas' ? dataRevistas : dataHistoricos;

    // Cambiamos el título del submodal
    document.getElementById('subModalTitle').textContent = category === 'revistas' ? "Revistas de la Agrupación" : "Archivo Histórico";

    // Generamos el HTML con la lista de documentos
    let html = '';
    items.forEach(item => {
      html += `
        <div class="bg-white dark:bg-gray-800 p-4 rounded-xl border border-agonia/10 hover:border-agonia cursor-pointer flex items-center gap-4 shadow-sm" 
             onclick="openPdfModal('${item.url}', '${item.title}', '${item.year || item.date}')">
          <span class="material-symbols-outlined text-agonia text-3xl">article</span>
          <div class="flex-1 truncate text-left">
            <h4 class="font-serif text-sm md:text-base truncate">${item.title}</h4>
            <p class="text-xs text-gray-500">${item.year || item.date}</p>
          </div>
        </div>
      `;
    });

    // Metemos el HTML en el contenedor
    document.getElementById('subModalContent').innerHTML = html;

    // Mostramos el submodal y bloqueamos el scroll
    document.getElementById('subModal').classList.add('active');
    document.body.style.overflow = 'hidden';
  }

  // Cierra el submodal
  function closeSubModal() {
    document.getElementById('subModal').classList.remove('active');
    document.body.style.overflow = '';
  }

  // Abre el visor de PDF con el documento seleccionado
  function openPdfModal(url, title, subtitle) {
    // Extraemos el ID del archivo de Google Drive
    const fileId = url.match(/\/d\/([a-zA-Z0-9_-]+)/)?.[1];

    // Si no hay ID, mostramos un mensaje
    if (!fileId) {
      alert('No se pudo cargar el documento. Verifica la URL.');
      return;
    }

    // Actualizamos el título del visor
    document.getElementById('pdfTitle').innerHTML = `${title} <span class="opacity-60 text-xs">(${subtitle})</span>`;

    // Cargamos el PDF en el iframe (preview de Google Drive)
    document.getElementById('pdfFrame').src = `https://drive.google.com/file/d/${fileId}/preview`;

    // Configuramos el enlace de descarga
    document.getElementById('pdfDownload').href = `https://drive.google.com/uc?export=download&id=${fileId}`;

    // Mostramos el esqueleto de carga
    document.getElementById('pdfSkeleton').style.display = 'flex';

    // Cuando el iframe termine de cargar, ocultamos el esqueleto
    document.getElementById('pdfFrame').onload = () => {
      document.getElementById('pdfSkeleton').style.display = 'none';
    };

    // Mostramos el modal
    document.getElementById('pdfModal').classList.add('active');
    document.body.style.overflow = 'hidden';
  }

  // Cierra el visor de PDF
  function closePdfModal() {
    document.getElementById('pdfModal').classList.remove('active');
    document.body.style.overflow = '';
    // Limpiamos el iframe para liberar memoria
    setTimeout(() => {
      document.getElementById('pdfFrame').src = '';
    }, 300);
  }

  // Alterna el modo pantalla completa del visor
  function toggleFullscreen() {
    const container = document.getElementById('pdfContainer');
    const btn = document.getElementById('fullscreenToggleBtn');

    if (!container.classList.contains('fullscreen-mode')) {
      container.classList.add('fullscreen-mode');
      btn.innerHTML = '<span class="material-symbols-outlined text-3xl">fullscreen_exit</span>';
    } else {
      container.classList.remove('fullscreen-mode');
      btn.innerHTML = '<span class="material-symbols-outlined text-3xl">fullscreen</span>';
    }
  }

  // ============================================================
  // EVENTOS: asignamos el clic a las tarjetas principales
  // ============================================================
  document.querySelectorAll('[data-modal]').forEach(card => {
    card.addEventListener('click', () => {
      // Extraemos la categoría del atributo data-modal (ej: "modal-revistas" → "revistas")
      const category = card.getAttribute('data-modal').split('-')[1];
      openSubModal(category);
    });
  });

  // ============================================================
  // CIERRE CON TECLA ESCAPE
  // ============================================================
  document.addEventListener('keydown', function(e) {
    if (e.key === 'Escape') {
      if (document.getElementById('pdfModal').classList.contains('active')) {
        closePdfModal();
      } else if (document.getElementById('subModal').classList.contains('active')) {
        closeSubModal();
      }
    }
  });
