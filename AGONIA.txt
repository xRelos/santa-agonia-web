<p>
    El tercio de la <strong class="text-agonia"> Santa Agonía</strong> procesiona en la noche del <strong>Viernes Santo</strong>, momento litúrgico que representa las últimas horas de Cristo en la cruz antes de su muerte. Esta procesión forma parte de la procesión del <strong>Santo Entierro</strong> y es una de las más emblemáticas de la Semana Santa de Cartagena, perteneciente a la Cofradía Marraja.
</p>
<p>
    La salida se realiza a las <strong>20:00 horas</strong> desde la <strong>Iglesia de Santa María de Gracia</strong>, situada en la <strong>Calle del Aire</strong>. El recorrido procesional atraviesa las principales calles del centro histórico de Cartagena, incluyendo la <strong>Calle Mayor</strong>, la <strong>Calle del Carmen</strong> y las <strong>Puertas de Murcia</strong>, finalizando en el mismo templo de origen tras aproximadamente cuatro horas de recorrido.
</p>
<p>
    Fundada en <strong>1929</strong>, este tercio procesiona las imagenes del <strong>Cristo de la  Agonía</strong>, obra del escultor catalán <strong>Carles Flotats i Galtés (1942)</strong>, y la <strong>Virgen de la Amargura</strong>, obra de <strong>Efraín Gómez Montón (1963)</strong>. Ambas imágenes representan el realismo sacro característico de la Cofradía Marraja.
</p>

// Variables para el carousel
let currentItem = 'imagen';
let currentImageIndex = 0;

// Elementos del DOM
const modal = document.getElementById('patrimonioModal');
const modalTitle = document.getElementById('modalTitle');
const modalContent = document.getElementById('modalContent');
const carouselImage = document.getElementById('carouselImage');
const imageCaption = document.getElementById('imageCaption');
const carouselIndicators = document.getElementById('carouselIndicators');
const closeModal = document.getElementById('closeModal');
const prevBtn = document.getElementById('prevBtn');
const nextBtn = document.getElementById('nextBtn');

// Funciones para manejar el modal
function openModal(itemId) {
    currentItem = itemId;
    currentImageIndex = 0;
    
    const data = patrimonioData[itemId];
    modalTitle.textContent = data.title;
    modalContent.innerHTML = data.content;
    
    // Cargar la primera imagen
    updateCarousel();
    
    // Mostrar modal
    modal.classList.add('active');
    document.body.style.overflow = 'hidden';
}

function closeModalFunc() {
    modal.classList.remove('active');
    document.body.style.overflow = 'auto';
}

function updateCarousel() {
    const data = patrimonioData[currentItem];
    const imageData = data.images[currentImageIndex];
    
    carouselImage.src = imageData.src;
    carouselImage.alt = imageData.caption;
    imageCaption.textContent = imageData.caption;
    
    // Actualizar indicadores
    carouselIndicators.innerHTML = '';
    data.images.forEach((_, index) => {
        const indicator = document.createElement('div');
        indicator.className = `w-3 h-3 rounded-full cursor-pointer ${index === currentImageIndex ? 'bg-agonia' : 'bg-gray-300 dark:bg-gray-700'}`;
        indicator.addEventListener('click', () => {
            currentImageIndex = index;
            updateCarousel();
        });
        carouselIndicators.appendChild(indicator);
    });
}

function nextImage() {
    const data = patrimonioData[currentItem];
    currentImageIndex = (currentImageIndex + 1) % data.images.length;
    updateCarousel();
}

function prevImage() {
    const data = patrimonioData[currentItem];
    currentImageIndex = (currentImageIndex - 1 + data.images.length) % data.images.length;
    updateCarousel();
}

// Event Listeners
document.querySelectorAll('.modern-bubble').forEach(bubble => {
    bubble.addEventListener('click', () => {
        const target = bubble.getAttribute('data-target');
        openModal(target);
    });
});

closeModal.addEventListener('click', closeModalFunc);
prevBtn.addEventListener('click', prevImage);
nextBtn.addEventListener('click', nextImage);

// Cerrar modal al hacer clic fuera del contenido
modal.addEventListener('click', (e) => {
    if (e.target === modal) {
        closeModalFunc();
    }
});

// Cerrar modal con tecla ESC
document.addEventListener('keydown', (e) => {
    if (e.key === 'Escape' && modal.classList.contains('active')) {
        closeModalFunc();
    }
});

// Añadir efecto de sonido sutil al hacer clic (opcional)
document.querySelectorAll('.modern-bubble').forEach(bubble => {
    bubble.addEventListener('click', () => {
        // Pequeña animación de pulso al hacer clic
        bubble.style.transform = 'scale(0.95)';
        setTimeout(() => {
            bubble.style.transform = '';
        }, 150);
    });
});
